List and Vector Poly Class

Testing:

In order to thoroughly test our Poly class as well as the List and Vector it uses, please use the following main.cpp.

main():

calls the test_poly_random() and exits.

test_poly_random():

test_poly_random generates two random polynomials and passes them to the display_results() function. The return value of this function is used to report the result of this test.

test_poly_random does this in a loop for 2000 times (obviously, you can change this number while debugging.) and will report the final result before exiting.

display_results():

Will take two polynomials (and an optional boolean) and will pass these arguments to the individual test functions for each of the four arithmetic operations. display_results() will return true if all of the arithmetic test functions return true indicating the operations were verified. returns false otherwise.

 

test_add(), test_sub(), test_mult(), test_div():

These are the individual arithmetic test functions. Each of these functions take in two polynomials and perform the arithmetic operation on them. They will then attempt to verify the correctness of the results. They all return a boolean indicating the success of the test.

 

Notes:

Poly CTOR:

The Poly constructor takes an array list of coeficients (in reverse order. i.e. the coef of the zero exponent is at array element zero) and an order that indicates the highest order (highest indes on the array.) Note that order is NOT the number of elements in the array but one less.

bool Poly::zero():

Add this function to your Poly class to return true if the all coeficients of all the terms are zero. To simplify matters, we check with this function and if the random poly is zero, we will not test this poly. This function is called from main and so it is required to run this test program.

int Poly::order():

I found that having an _order private member variable in Poly requires careful inspection of every function to make sure the poly's highest order terms are not zero (which is why we needed the fix_order() function.) In the interest of time, I decided to remove this member variable and instead call the order() function. This function is not called from main() so it is not a requirement but a recommendation. 

The order() function will return the real order of the polynomial:

[0x^5 + 0x^4+8x^3...] will have an order of 3

void test_poly_random(bool report){
    bool verified_all = true;
    bool verified = true;
    string verification_str;
    Random r;
    Poly p;
    Poly q;
    double p_array[20];

    for(int iterations = 0; iterations <2000; iterations++){

    cout<<"******************************************************************************"<<endl;
    cout<<"******************************************************************************"<<endl;
}

bool display_results(const Poly& p, const Poly& q, bool report){
    bool verified = true;
    if (report) cout<<endl<<"------- p + q ----------------------"<<endl;
    verified = test_add(p, q, report) && verified;
    if (report) cout<<endl<<"------- q + p ----------------------"<<endl;
    verified = test_add(q, p, report) && verified;
    if (report) cout<<endl<<"------- p - q ----------------------"<<endl;
    verified = test_sub(p, q, report) && verified;
    if (report) cout<<endl<<"------- q - p ----------------------"<<endl;
    verified = test_sub(q, p, report) && verified;
    if (report) cout<<endl<<"------- p * q ----------------------"<<endl;
    verified = test_mult(p, q, report) && verified;
    if (report) cout<<endl<<"------- q * p ----------------------"<<endl;
    verified = test_mult(q, p, report) && verified;
    if (report) cout<<endl<<"------- p / q ----------------------"<<endl;
    verified = test_div(p, q, report) && verified;
    if (report) cout<<endl<<"------- q / p ----------------------"<<endl;
    verified = test_div(q, p, report) && verified;
    return verified;
}
bool test_mult(const Poly& p, const Poly& q, bool report){
    Poly r = p * q;
    if(report) cout<<p<<" * "<<q<<endl;
    if(report) cout<<"      = "<<r<<"  ";
    //if r = p*q then r/p +r%p should be equal to q:
    if(r / p + r % p == q){
        if(report) cout<<"VERIFIED"<<endl;
        return true;
    }
    else{
        if(report) cout<<endl<<"   ***VERIFICATION FAILED"<<endl;
        if(report) cout<<"   r / p: "<<r / p<<endl;
        if(report) cout<<"   r % p: "<<r / p<<endl;
        if(report) cout<<"   r / p + r % p: "<<r / p + r % p<<endl;
        return false;
    }

}
bool test_div(const Poly& p, const Poly& q, bool report){
    Poly r = p / q;
    if(report) cout<<p<<" / "<<q<<endl;
    if(report) cout<<"      = "<<r;
    if(report) cout<<"  rem = "<<p % q<<"  ";
    //if r = p / q then r * q + p % q should be equal to p:
    if(r * q + p % q == p){
        if(report) cout<<"VERIFIED"<<endl;
        return true;
    }
    else{
        if(report) cout<<endl<<"   ***VERIFICATION FAILED"<<endl;
        if(report) cout<<"   r * q: "<<r * q<<endl;
        if(report) cout<<"   p % q: "<<p % q<<endl;
        if(report) cout<<"   r * q + p % q: "<<r * q + p % q<<"=/="<<p<<endl;
        return false;
    }
}

bool test_add(const Poly& p, const Poly& q, bool report){
    const bool debug = false;
    Poly r = p + q;
    if(report) cout<<"["<<p<<"] + ["<<q<<"]"<<endl;
    if(report) cout<<"      = "<<r<<"  ";
    if (debug) cout<<endl<<"----(test_add) VERIFICATION: ------"<<endl;
    if (debug) cout<<"checking if "<<r<<" - "<<p<<" (diff) is equal to"<<endl
                  <<"      "<<q<<"??"<<endl;
    Poly diff = r-p;
    if (debug){
        cout<<"------- diff ------------"<<endl;
        cout<<endl<<"diff: "<<diff<<endl;
        cout<<"-------------------------"<<endl;
    }
    if (diff == q){
        if(report) cout<<"VERIFIED "<<endl;
        return true;
    }
    else{
        if(report) cout<<endl<<"   ***VERIFICATION FAILED"<<endl;
        if(report) cout<<"result - p = ["<<diff<<"]"<<endl;
        return false;
    }
}
bool test_sub(const Poly& p, const Poly& q, bool report){
    const bool debug = false;
    Poly r = p - q;
    if (report) cout<<p<<" - "<<q<<endl;
    if (report) cout<<"      = "<<r<<"  ";
    if (debug) cout<<endl<<"----(test_sub) VERIFICATION: ------"<<endl;
    if (debug) cout<<"checking if "<<r<<" + "<<q<<" (total) is equal to"<<endl
                  <<"      "<<p<<endl;
    Poly total = r + q;
    if (debug){
        cout<<"p order: "<<p<<", order: "<<p.order()<<endl;
        cout<<"total order: "<<total<<", order: "<<total.order()<<endl;
    }
    if (total == p){
        if (report) cout<<"VERIFIED "<<endl;
        return true;
    }
    else{
        if (report) cout<<endl<<"   ***VERIFICATION FAILED"<<endl;
        if (report) cout<<"result + q = "<<total<<" != "<<p<<endl;
        return false;
    }
}

 

class Random:

random.h

#ifndef RANDOM_H
#define RANDOM_H

class Random
{
public:
    Random();
    int GetNext(int lo, int hi);
private:
    //----------------------------------------------------------------------
    //static member variable keeps the random number generator from being seeded everytime a
    // new Random object is instantiated.
    //Note the initialization of the static variable in the .cpp file.
    static int objectCount;
    //----------------------------------------------------------------------
};

#endif // RANDOM_H

random.cpp:

#include "random.h"
#include <time.h>
#include <stdlib.h>
#include <iostream>

using namespace std;

//----------------------------------------------------------------------
//initialization of the static member variable must be done in the .cpp file.
//
int Random::objectCount=0;
//
//----------------------------------------------------------------------


Random::Random(){
    objectCount++;
    if (objectCount==1){
        //cout<<endl<<"random number generator seeded..."<<endl;
        //-----------------------------------------------
        srand(time(0)); //this is bad!!//
    }
}

int Random::GetNext(int lo, int hi){
    int r = rand()%(hi-lo+1)+lo;
    return r;

}
